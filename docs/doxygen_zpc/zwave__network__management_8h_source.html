336/* An application MUST time out waiting for the
337 * ADD_NODE_STATUS_NODE_FOUND status if it does not receive the indication
338 * after calling AddNodeToNetwork(ADD_NODE_ANY).
339 * Value: 60 seconds (60000 ms) Refer: Z-wave Appl. Programmers Guide */

345#define SET_DEFAULT_TIMEOUT 15000

348#define LEARN_MODE_TIMEOUT 30000
349
352#define SEND_DATA_TIMEOUT 125000

354/* An application MUST time out waiting for the ADD_NODE_STATUS_ADDING_END_NODE
355 * status if it does not receive the indication within a time period after
356 * receiving the ADD_NODE_STATUS_NODE_FOUND status.
357 * The RECOMMENDED timeout interval is 60 sec. */
358#define ADD_NODE_STATUS_ADDING_END_NODE_TIMEOUT 60000
359
360/*
361 * INS13954 indicates that the timeout should be as follows:
362 * 1. for including an End Node:
363 * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms
364 * 2. for including a Controller Node:
365 * timeout = 76000 ms + listening nodes * 217 ms + FLiRS Nodes * 3517 ms + network nodes * 732 ms
366 *
367 * This is the worst case scenario time out, we do not bother calculating the right timeout
368 */
369#define ADD_NODE_PROTOCOL_NEIGHBOR_DISCOVERY_TIMEOUT \
370 (76000 + ZW_MAX_NODES * (3517 + 732))
